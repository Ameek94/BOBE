 42def dist_sq(x, y):
 43    """
 44    Compute squared Euclidean distance between two points x, y. 
 45    If x is n1 x d and y is n2 x d returns a n1 x n2 matrix of distancess.
 46    """
 47    return jnp.sum(jnp.square(x[:,None,:] - y),axis=-1)

 49@partial(jit,static_argnames='include_noise')
 50def rbf_kernel(xa,
 51               xb,
 52               lengthscales,
 53               outputscale,
 54               noise,include_noise=True): 
 55    """
 56    The RBF kernel
 57    """
 58    sq_dist = dist_sq(xa/lengthscales,xb/lengthscales) 
 59    sq_dist = jnp.exp(-0.5*sq_dist)
 60    k = outputscale*sq_dist
 61    if include_noise:
 62        k+= noise*jnp.eye(k.shape[0])
 63    return k

 65@partial(jit,static_argnames='include_noise')
 66def matern_kernel(xa,xb,lengthscales,outputscale,noise,include_noise=True):
 67    """
 68    The Matern-5/2 kernel
 69    """
 70    dsq = dist_sq(xa/lengthscales,xb/lengthscales)
 71    d = jnp.sqrt(jnp.where(dsq<1e-30,1e-30,dsq))
 72    exp = jnp.exp(-sqrt5*d)
 73    poly = 1. + d*(sqrt5 + d*5./3.)
 74    k = outputscale*poly*exp
 75    if include_noise:
 76        k+= noise*jnp.eye(k.shape[0])
 77    return k

 79@jit
 80def get_var_from_cho(k11_cho,k12,k22):
 81    vv = solve_triangular(k11_cho,k12,lower=True)
 82    var = jnp.diag(k22) - jnp.sum(vv*vv,axis=0) # replace k22 computation with single element diagonal
 83    return var

428    def add(self,new_x,new_y):
429        """
430        Updates the GP with new training points.
431        """
432        self.train_x = jnp.concatenate([self.train_x,new_x])
433        self.train_y = self.train_y*self.y_std + self.y_mean
434        self.train_y = jnp.concatenate([self.train_y,new_y])
435        self.y_mean = jnp.mean(self.train_y,axis=0)
436        self.y_std = jnp.std(self.train_y,axis=0)
437        self.train_y = (self.train_y - self.y_mean) / self.y_std

464    def predict_mean(self,x):
465        x = jnp.atleast_2d(x)
466        k12s = self.get_kernels(self.train_x,x,include_noise=False)
467        mean_f = lambda cho, k12 : (get_mean_from_cho(cho,k12,self.train_y),)
468        means = split_vmap(mean_f,input_arrays=(self.cholesky,k12s),batch_size=self.vmap_size)[0]
469        return means.mean(axis=0) * self.y_std + self.y_mean

471    def predict_var(self,x):
472        x = jnp.atleast_2d(x)
473        k12s = self.get_kernels(self.train_x,x,include_noise=False)
474        k22s = self.get_kernels(x,x,include_noise=True)
475        var_f = lambda cho, k12, k22 : (get_var_from_cho(cho,k12,k22),)
476        vars = split_vmap(var_f,input_arrays=(self.cholesky,k12s,k22s),batch_size=self.vmap_size)[0]
477        return vars.mean(axis=0)*self.y_std**2
